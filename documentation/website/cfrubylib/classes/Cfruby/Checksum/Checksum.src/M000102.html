                        def databasefile=(newdatabasefile)
                                @databasefile = newdatabasefile

                                if(!File.exists?(@databasefile))
                                        save_database()
                                end
                                open_database()
                                
                                Cfruby.controller.inform('debug', "checksum databasefile set to \"#{@databasefile}\"")
                        end
