                        def open_database()
                                # throw an exception if the mode isn't right
                                check_mode()

                                Cfruby.controller.inform('debug', "opening checksum database \"#{@databasefile}\"")
                                File.open(@databasefile, File::RDONLY) { |fp|
                                        @database = YAML.load(fp)
                                }
                        end
