                        def calculate_bump(stack)
                                calls = 0
                                last = ''
                                stack.each() { |s|
                                        if(s =~ /\/flowmonitor\.rb:/)
                                                if(last !~ /\/flowmonitor\.rb:/)
                                                        calls = calls + 1
                                                end
                                        end
                                        last = s
                                }
                                
                                return(calls)
                        end
