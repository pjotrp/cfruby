                def FileEdit.convert_to_fuzzy_regex(str, options = {})
                        tokens = str.strip.split(/\s+/).map { |tok|
                                Regexp.escape(tok)
                        }
                        
                        regex = tokens.join("\\s+")
                        if(!options[:partialstr])
                                regex = "^\\s*#{regex}\\s*$"
                        end
                        regex = Regexp.new(regex)
                        
                        Cfruby.controller.inform('debug', "\"#{str}\" converted to \"#{regex.source()}\"")
                        return(regex)
                end
