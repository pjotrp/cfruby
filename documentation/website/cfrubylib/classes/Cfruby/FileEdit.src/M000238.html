                def FileEdit.file_append(filename, str) 
                        changed = false
                        Cfruby.controller.attempt("appending \"#{str}\" to #{filename}", 'destructive') {
                                changed = FileEdit.open(filename, File::WRONLY|File::APPEND) { |fp|
                                        fp.write(str)
                                }
                        }
                        
                        return(changed)
                end
