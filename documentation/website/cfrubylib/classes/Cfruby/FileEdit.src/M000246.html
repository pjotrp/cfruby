                def FileEdit.write_file(filename, lines)
                        changed = FileEdit.open(filename, File::WRONLY) { |fp|
                                fp.truncate(0)
                                lines.each() { |line|
                                        fp.write(line)
                                }
                        }
                        
                        return(changed)
                end
