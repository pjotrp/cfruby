                        def group?(group)
                                if(groups.has_key?(group))
                                        Cfruby.controller.inform('debug', "group \"#{group}\" exists")
                                        return(true)
                                end

                                Cfruby.controller.inform('debug', "group \"#{group}\" does not exist")
                                return(false)
                        end
