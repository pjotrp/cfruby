                        def infile(str, filename)
                                regex = Regexp.new("^#{Regexp.escape(str)}:")
                                File.open(filename, File::RDONLY) { |fp|
                                        fp.each_line() { |line|
                                                if(regex.match(line))
                                                        return(true)
                                                end
                                        }
                                }

                                return(false)
                        end
