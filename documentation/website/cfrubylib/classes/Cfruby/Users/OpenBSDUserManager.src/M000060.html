                        def user?(user)
                                username = ""
                                if(user.respond_to?(:username))
                                        username = user.username
                                else
                                        username = user
                                end

                                output = Exec::exec("/usr/sbin/pw showuser '#{shellescape(username)}'")
                                if(output[0][0] =~ /^#{Regexp.escape(username)}:/)
                                        return(true)
                                else
                                        return(false)
                                end
                        end
