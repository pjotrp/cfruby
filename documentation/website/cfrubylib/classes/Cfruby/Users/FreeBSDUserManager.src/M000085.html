                        def delete_user(user, removehome=false)
                                username = nil
                                if(user.respond_to?(:username))
                                        username = user.username
                                else
                                        username = user.to_s
                                end
                                Cfruby.controller.attempt("Removing user \"#{username}\"", 'nonreversible', 'destructive') {
                                        if(removehome == true)
                                                `pw userdel #{username} -r`
                                        else
                                                `pw userdel #{username}`
                                        end
                                }
                        end
