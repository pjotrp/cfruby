                        def user?(user)
                                if(users()[user] != nil)
                                        Cfruby.controller.inform('debug', "\"#{user}\" exists")
                                        return(true)
                                else
                                        Cfruby.controller.inform('debug', "\"#{user}\" does not exist")
                                        return(false)
                                end
                        end
