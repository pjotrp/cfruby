                        def installed?(package)
                                Cfruby.controller.inform('debug', "Getting installed? status of \"#{package}\" by origin")

                                installed_packages.each_value() { |packageinfo|
                                        if(packageinfo.origin == package)
                                                return(true)
                                        end
                                }

                                return(super(package))
                        end
