                        def packages()
                                packages = PackageList.new()
                                packageregex = /^\s*([^\/]+)\/([^ ]+?)-([0-9][^ ]+)/

                                list = `qpkg -v`
                                list.each_line() { |line|
                                        match = packageregex.match(line)
                                        if(match != nil)
                                                packages[match[2]] = PackageInfo.new()
                                                packages[match[2]].name = match[2]
                                                packages[match[2]].version = match[3]
                                                packages[match[2]].category = match[1]
                                        end
                                }
                        end
