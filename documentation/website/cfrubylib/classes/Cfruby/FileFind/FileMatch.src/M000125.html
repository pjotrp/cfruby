                        def empty(filename)
                                test = Pathname.new(filename)
                                if(test.directory?)
                                        count = 0
                                        test.each_entry { |d|
                                                if(d.to_s !~ /^\.+$/)
                                                        count = count + 1
                                                end
                                        }
                                        if(count == 0)
                                                return(true)
                                        end
                                elsif(test.symlink?)
                                        begin
                                                test.realpath
                                        rescue
                                                if($!.to_s() =~ /No such file or directory/)
                                                        return(true)
                                                else
                                                        raise($!)
                                                end
                                        end
                                elsif(test.file?)
                                        if(test.size == 0)
                                                return(true)
                                        end
                                end
                                
                                return(false)
                        end
