                        def convertsize(str)
                                if(str.kind_of?(Numeric))
                                        return(str)
                                end
                                
                                str.strip!()
                                regex = /^([0-9]+(?:\.[0-9]+)?)\s*([gmk])b?$/i
                                match = regex.match(str)
                                if(match == nil)
                                        raise(ArgumentError, "Unable to parse size of \"#{str}\"")
                                end
                                
                                size = match[1].to_f()
                                
                                case(match[2].downcase())
                                when('g')
                                        size = size * 1024 * 1024 * 1024
                                when('m')
                                        size = size * 1024 * 1024
                                when('k')
                                        size = size * 1024
                                end
                                
                                return(size)
                        end
