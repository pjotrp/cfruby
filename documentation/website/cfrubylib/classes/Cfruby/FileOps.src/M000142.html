                def FileOps.disable(basedir, options = {})
                  disabled = false
                        Cfruby.controller.attempt("disabling file in \"#{basedir}\"", 'destructive') {
                                Cfruby::FileFind.find(basedir, options) { |filename|
                                        if(Cfruby::FileOps.chmod(filename, 0000))
                                          disabled = true
                                  end
                                }
                        }
                        
                        return(disabled)
                end
