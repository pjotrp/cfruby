                def FileOps.delete_nonalpha(basedir, options = {})
                  deleted = false
                        Cfruby.controller.attempt("deleting files non-alpha files from \"#{basedir}\"", 'nonreversible', 'destructive') {
                                if(FileOps.delete_not_matching_regex(basedir, /[a-zA-Z0-9]/))
                                  deleted = true
                          end
                        }
                        
                        return(deleted)
                end
