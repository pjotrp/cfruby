        def CfYAML.load(filename, changed=false)
                fp = filename
                if(!filename.kind_of?(IO))
                        fp = File.open(filename, File::RDONLY)
                end
                
                cfyamlcode = YAML.load(fp)

                return(process(cfyamlcode, changed))
        end
